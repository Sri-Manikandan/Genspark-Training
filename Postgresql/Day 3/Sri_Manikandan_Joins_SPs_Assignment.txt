Transactions and SPs

Q1. Create a stored procedure to insert a new customer. Use a transaction so that if any required value is missing, the insert is rolled back.
1.
create or replace procedure insert_customer(p_customerid CHAR(5),p_companyname VARCHAR(40),p_contactname VARCHAR(30),p_contacttitle VARCHAR(30),p_address VARCHAR(60),p_city VARCHAR(15),p_region VARCHAR(15),p_postalcode VARCHAR(10),p_country VARCHAR(15),p_phone VARCHAR(24),p_fax VARCHAR(24))
language plpgsql
as $$
begin
	-- START TRANSACTION;
	IF p_customerid is null  or trim(p_customerid)=''or
		p_companyname is null or trim(p_companyname)=''then
		-- ROLLBACK;
		RAISE EXCEPTION 'CustomerID or Company name is required';
	END IF;
	INSERT INTO customers (customerid, companyname, contactname, contacttitle, address, city, region, postalcode, country, phone, fax) VALUES (p_customerid, p_companyname, p_contactname, p_contacttitle, p_address, p_city, p_region, p_postalcode, p_country, p_phone, p_fax);
	-- COMMIT;
end;
$$;

CALL insert_customer('C1004', 'Synergeek Technologies', 'Manish', 'CEO', '123 Main Road', 'Coimbatore', NULL, '641002', 'India', '9876543210', NULL);

Q2. Create a stored procedure to place a new order for an existing customer with one product. Insert into orders and order_details in a single transaction.
2.
CREATE OR REPLACE PROCEDURE place_order(
p_customerid TEXT,
p_employeeid INT,
p_productid INT,
p_quantity INT
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_orderid INT;
    v_unitprice NUMERIC(19,4);
BEGIN
	-- START TRANSACTION;
    IF p_customerid IS NULL OR TRIM(p_customerid) = '' THEN
		-- ROLLBACK;
        RAISE EXCEPTION 'CustomerID is required';
    END IF;

    IF p_productid IS NULL OR p_quantity <= 0 THEN
		-- ROLLBACK;
        RAISE EXCEPTION 'Valid product and quantity required';
    END IF;

    SELECT unitprice INTO v_unitprice
    FROM products
    WHERE productid = p_productid;

    IF v_unitprice IS NULL THEN
		-- ROLLBACK;
        RAISE EXCEPTION 'Product not found';
    END IF;

    INSERT INTO orders (customerid,employeeid,orderdate,freight)
    VALUES (p_customerid,p_employeeid,CURRENT_TIMESTAMP,0)
    RETURNING orderid INTO v_orderid;

    INSERT INTO order_details (orderid,productid,unitprice,quantity,discount)
    VALUES (v_orderid,p_productid,v_unitprice,p_quantity,0);
	-- COMMIT;
END;
$$;

CALL place_order('ALFKI', 1, 1, 5);

Q3. Create a stored procedure to update product stock after an order is placed. If stock is not enough, rollback the transaction.
3.
CREATE OR REPLACE PROCEDURE update_product_stock(p_productid INT, p_quantity INT)
LANGUAGE plpgsql
AS $$
DECLARE
    v_stock INT;
BEGIN
    -- START TRANSACTION;
    SELECT unitsinstock INTO v_stock FROM products WHERE productid = p_productid;
    IF v_stock IS NULL THEN
        -- ROLLBACK;
        RAISE EXCEPTION 'Product not found';
    END IF;
    IF v_stock < p_quantity THEN
        -- ROLLBACK;
        RAISE EXCEPTION 'Insufficient stock. Available: %, Requested: %', v_stock, p_quantity;
    END IF;
    UPDATE products SET unitsinstock = unitsinstock - p_quantity WHERE productid = p_productid;
    -- COMMIT;
END;
$$;

CALL update_product_stock(1, 10);

Q4. Create a stored procedure to cancel an order. Delete records from order_details first, then from orders, using a transaction.
4.
CREATE OR REPLACE PROCEDURE cancel_order(p_orderid INT)
LANGUAGE plpgsql
AS $$
BEGIN
    -- START TRANSACTION;
    IF NOT EXISTS (SELECT 1 FROM orders WHERE orderid = p_orderid) THEN
        -- ROLLBACK;
        RAISE EXCEPTION 'Order not found';
    END IF;
    DELETE FROM order_details WHERE orderid = p_orderid;
    DELETE FROM orders WHERE orderid = p_orderid;
    -- COMMIT;
END;
$$;

CALL cancel_order(10248);

Q5. Create a stored procedure to transfer products from one supplier to another. If the old supplier or new supplier does not exist, rollback.
5.
CREATE OR REPLACE PROCEDURE transfer_supplier(p_old_supplierid INT, p_new_supplierid INT)
LANGUAGE plpgsql
AS $$
BEGIN
    -- START TRANSACTION;
    IF NOT EXISTS (SELECT 1 FROM suppliers WHERE supplierid = p_old_supplierid) THEN
        -- ROLLBACK;
        RAISE EXCEPTION 'Old supplier not found';
    END IF;
    IF NOT EXISTS (SELECT 1 FROM suppliers WHERE supplierid = p_new_supplierid) THEN
        -- ROLLBACK;
        RAISE EXCEPTION 'New supplier not found';
    END IF;
    UPDATE products SET supplierid = p_new_supplierid WHERE supplierid = p_old_supplierid;
    -- COMMIT;
END;
$$;

CALL transfer_supplier(1, 2);

Q6. Create a stored procedure to update the price of all products in a category by a percentage. Rollback if the percentage is less than or equal to zero.
6.
CREATE OR REPLACE PROCEDURE update_category_price(p_categoryid INT, p_percentage NUMERIC)
LANGUAGE plpgsql
AS $$
BEGIN
    -- START TRANSACTION;
    IF p_percentage <= 0 THEN
        -- ROLLBACK;
        RAISE EXCEPTION 'Percentage must be greater than zero';
    END IF;
    IF NOT EXISTS (SELECT 1 FROM categories WHERE categoryid = p_categoryid) THEN
        -- ROLLBACK;
        RAISE EXCEPTION 'Category not found';
    END IF;
    UPDATE products SET unitprice = unitprice * (1 + p_percentage / 100) WHERE categoryid = p_categoryid;
    -- COMMIT;
END;
$$;

CALL update_category_price(1, 10);

LANGUAGE plpgsql

    -- START TRANSACTION;

Q8. Create a stored procedure to delete a customer only if the customer has no orders. Use a transaction and rollback if orders exist.
8.
CREATE OR REPLACE PROCEDURE delete_customer(p_customerid CHAR(5))
LANGUAGE plpgsql
AS $$
BEGIN
    -- START TRANSACTION;
    IF NOT EXISTS (SELECT 1 FROM customers WHERE customerid = p_customerid) THEN
        -- ROLLBACK;
        RAISE EXCEPTION 'Customer not found';
    END IF;
    IF EXISTS (SELECT 1 FROM orders WHERE customerid = p_customerid) THEN
        -- ROLLBACK;
        RAISE EXCEPTION 'Cannot delete customer with existing orders';
    END IF;
    DELETE FROM customers WHERE customerid = p_customerid;
    -- COMMIT;
END;
$$;

Q10. Create a stored procedure to place an order with multiple products. Insert the order and all order items in one transaction. If any product is invalid or stock is insufficient, rollback the complete order.
10.
CREATE OR REPLACE PROCEDURE place_order_multi(p_customerid TEXT, p_employeeid INT, p_productids INT[], p_quantities INT[])
LANGUAGE plpgsql
AS $$
DECLARE
    v_orderid INT;
    v_unitprice NUMERIC(19,4);
    v_stock INT;
    i INT;
BEGIN
    -- START TRANSACTION;
    IF p_customerid IS NULL OR TRIM(p_customerid) = '' THEN
        -- ROLLBACK;
        RAISE EXCEPTION 'CustomerID is required';
    END IF;
    IF NOT EXISTS (SELECT 1 FROM customers WHERE customerid = p_customerid) THEN
        -- ROLLBACK;
        RAISE EXCEPTION 'Customer not found';
    END IF;
    IF array_length(p_productids, 1) IS NULL THEN
        -- ROLLBACK;
        RAISE EXCEPTION 'At least one product is required';
    END IF;
    FOR i IN 1..array_length(p_productids, 1) LOOP
        SELECT unitprice, unitsinstock INTO v_unitprice, v_stock FROM products WHERE productid = p_productids[i];
        IF v_unitprice IS NULL THEN
            -- ROLLBACK;
            RAISE EXCEPTION 'Product % not found', p_productids[i];
        END IF;
        IF p_quantities[i] <= 0 THEN
            -- ROLLBACK;
            RAISE EXCEPTION 'Quantity for product % must be greater than zero', p_productids[i];
        END IF;
        IF v_stock < p_quantities[i] THEN
            -- ROLLBACK;
            RAISE EXCEPTION 'Insufficient stock for product %. Available: %, Requested: %', p_productids[i], v_stock, p_quantities[i];
        END IF;
    END LOOP;
    INSERT INTO orders (customerid, employeeid, orderdate, freight)
    VALUES (p_customerid, p_employeeid, CURRENT_TIMESTAMP, 0)
    RETURNING orderid INTO v_orderid;
    FOR i IN 1..array_length(p_productids, 1) LOOP
        SELECT unitprice INTO v_unitprice FROM products WHERE productid = p_productids[i];
        INSERT INTO order_details (orderid, productid, unitprice, quantity, discount)
        VALUES (v_orderid, p_productids[i], v_unitprice, p_quantities[i], 0);
        UPDATE products SET unitsinstock = unitsinstock - p_quantities[i] WHERE productid = p_productids[i];
    END LOOP;
    -- COMMIT;
END;
$$;

CALL place_order_multi('ALFKI', 1, ARRAY[1, 2, 3], ARRAY[5, 3, 2]);
